JavaScript - DOM Methods

Q1. document.getElementById()

  • Description: Selects a single element by its ID.
  • Syntax: document.getElementById("#id")
  • Parameters: id (string) - The ID of the element to find.
  • Return Value: The element object or null if no element with the specified ID is found.
                        
                            var element = document.getElementById('myElement');
                            console.log(element);

Q2. document.querySelector()

  • Description: Selects the first element that matches a CSS selector.
  • Syntax: document.querySelector(selector)
  • Parameters: `selector` (string) - A CSS selector string.
  • Return Value: The first matching element or null if no match is found.
                    
                        var element = document.querySelector('.myClass');
                        console.log(element);

Q3. document.querySelectorAll()

  • Description: Selects all elements that match a CSS selector.
  • Syntax: document.querySelectorAll(selector)
  • Parameters: selector (string) - A CSS selector string.
  • Return Value: A NodeList of matching elements.
                    
                        var elements = document.querySelectorAll('.myClass');
                        elements.forEach(element => console.log(element));

Q4. document.createElement()

  • Description: Creates a new HTML element.
  • Syntax: document.createElement(tagName)
  • Parameters: tagName (string) - The name of the element to create.
  • Return Value: A new element object.
                    
                        var newDiv = document.createElement('div');
                        console.log(newDiv);

Q5. document.createTextNode()

  • Description: Creates a new text node.
  • Syntax: document.createTextNode(text)
  • Parameters: text (string) - The text content for the text node.
  • Return Value: A new text node object.
                    
                        var textNode = document.createTextNode('Hello, world!');
                        console.log(textNode);

Q6. element.appendChild()

  • Description: Appends a node as the last child of an element.
  • Syntax: element.appendChild(node)
  • Parameters: node - The node to append.
  • Return Value: The appended node.
                    
                        var parent = document.getElementById('parentElement');
                        var child = document.createElement('div');
                        parent.appendChild(child);

Q7. element.removeChild()

  • Description: Removes a child node from an element.
  • Syntax: element.removeChild(node)
  • Parameters: node - The child node to remove.
  • Return Value: The removed node.
                    
                        var parent = document.getElementById('parentElement');
                        var child = document.getElementById('childElement');
                        parent.removeChild(child);

Q8. element.replaceChild()

  • Description: Replaces a child node with a new node.
  • Syntax: element.replaceChild(newNode, oldNode)
  • Parameters: newNode - The node to insert.
    oldNode - The node to replace.
  • Return Value: The replaced node.
                    
                        var parent = document.getElementById('parentElement');
                        var newChild = document.createElement('div');
                        var oldChild = document.getElementById('childElement');
                        parent.replaceChild(newChild, oldChild);

Q9. element.insertBefore()

  • Description: Inserts a new node before a reference node as a child of a specified element.
  • Syntax: element.insertBefore(newNode, referenceNode)
  • Parameters:' newNode' - The node to insert.
    'referenceNode' - The node before which the new node will be inserted.
  • Return Value: The inserted node.
                    
                        var parent = document.getElementById('parentElement');
                        var newChild = document.createElement('div');
                        var referenceChild = document.getElementById('referenceElement');
                        parent.insertBefore(newChild, referenceChild);

Q 10. element.cloneNode()

  • Description: Clones a node.
  • Syntax: element.cloneNode(deep)
  • Parameters: 'deep' (boolean) - If 'true', clones the node and all its descendants; if false, clones only the node.
  • Return Value: The cloned node.
                    
                        var original = document.getElementById('originalElement');
                        var clone = original.cloneNode(true);
                        document.body.appendChild(clone);

Q 11. element.getAttribute()

  • Description: Gets the value of an attribute on an element.
  • Syntax: element.getAttribute(name)
  • Parameters: 'name' (string) - The name of the attribute.
  • Return Value: The value of the attribute or null if the attribute does not exist.
                    
                        var element = document.getElementById('myElement');
                        var value = element.getAttribute('data-custom');
                        console.log(value);

Q 12. element.setAttribute()

  • Description: Sets the value of an attribute on an element.
  • Syntax: element.setAttribute(name, value)
  • Parameters: name (string) - The name of the attribute.
  • Return Value: The value to set for the attribute.
                    
                        var element = document.getElementById('myElement');
                        element.setAttribute('data-custom', 'someValue');

Q 13. element.removeAttribute()

  • Description: Removes an attribute from an element.
  • Syntax: element.removeAttribute(name)
  • Parameters: name (string) - The name of the attribute to remove.
  • Return Value: None
                    
                        var element = document.getElementById('myElement');
                        element.removeAttribute('data-custom');

Q 14. element.classList.add()

  • Description: Adds one or more class names to an element.
  • Syntax: element.classList.add(class1, class2, ...)
  • Parameters: One or more class names (strings) to add to the element.
  • Return Value: None
                    
                        var element = document.getElementById('myElement');
                        element.classList.add('newClass');

Q 15. element.classList.remove()

  • Description: Removes one or more class names from an element.
  • Syntax: element.classList.remove(class1, class2, ...)
  • Parameters: One or more class names (strings) to remove from the element.
  • Return Value: None
                    
                        var element = document.getElementById('myElement');
                        element.classList.remove('oldClass');

Q 16. element.classList.toggle()

  • Description: Toggles a class name on an element.
  • Syntax: element.classList.toggle(className, force)
  • Parameters:
    'className (string)' - The class name to toggle.
    'force (boolean, optional)' - If true, adds the class; if false, removes it.
  • Return Value: true if the class is now present, false if it is now absent.
                    
                        var element = document.getElementById('myElement');
                        element.classList.toggle('active');

Q 17. element.classList.contains()

  • Description: Checks if an element has a specific class.
  • Syntax: element.classList.contains(className)
  • Parameters: className (string) - The class name to check for
  • Return Value: true if the class is present, false otherwise.
                    
                        var element = document.getElementById('myElement');
                        var hasClass = element.classList.contains('active');
                        console.log(hasClass);

Q 18. element.addEventListener()

  • Description: Adds an event listener to an element.
  • Syntax: element.addEventListener(type, listener, options)
  • Parameters:
    'type' (string) - The type of event to listen for (e.g., "click"). 'listener' (function) - The function to call when the event occurs. 'options' (object, optional) - Additional options (e.g., { once: true }).
  • Return Value: None
                    
                        var element = document.getElementById('myElement');
                        element.addEventListener('click', function() {
                            console.log('Element clicked!');
                        });

Q 19. element.removeEventListener()

  • Description: Removes an event listener from an element.
  • Syntax: element.removeEventListener(type, listener, options)
  • Parameters:
    'type' (string) - The type of event to stop listening for.
    'listener' (function) - The function to remove.
    'options' (object, optional) - Additional options (e.g., { once: true }).
  • Return Value: None
                    
                        var element = document.getElementById('myElement');
                        var handleClick = function() {
                            console.log('Element clicked!');
                        };
                        element.addEventListener('click', handleClick);
                        // Later...
                        element.removeEventListener('click', handleClick);

Q 20. element.dispatchEvent()

  • Description: Dispatches an event to an element.
  • Syntax: element.dispatchEvent(event)
  • Parameters: The event object to dispatch.
  • Return Value: 'true' if the event was not canceled, false otherwise. 
                    
                        var element = document.getElementById('myElement');
                        var event = new Event('build');
                        element.addEventListener('build', function(e) {
                            console.log('Event triggered!');
                        });
                        element.dispatchEvent(event);

Q 21. element.focus()

  • Description: Sets focus on the specified element.
  • Syntax: element.focus()
  • Parameters: None
  • Return Value: None
                    
                        var input = document.getElementById('myInput');
                        input.focus();

Q 22. element.blur()

  • Description: Removes focus from the specified element.
  • Syntax: element.blur()
  • Parameters: None
  • Return Value: None
                    
                        var input = document.getElementById('myInput');
                        input.blur();

Q 23. document.createEvent()

  • Description: Creates a new event object.
  • Syntax: document.createEvent(eventInterface)
  • Parameters: 'eventInterface' (string) - The type of event interface to create (e.g., "Event", "MouseEvent").
  • Return Value: The newly created event object
                    
                        var event = document.createEvent('Event');
                        event.initEvent('build', true, true);
                        var element = document.getElementById('myElement');
                        element.dispatchEvent(event);

Q 24. document.createRange()

  • Description: Creates a new Range object.
  • Syntax: document.createRange()
  • Parameters: None
  • Return Value: A new Range object.
                    
                        var range = document.createRange();
                        var element = document.getElementById('myElement');
                        range.selectNode(element);
                        console.log(range);

Q 25. document.execCommand()

  • Description: Executes a command for editable content.
  • Syntax: document.execCommand(command, showUI, value)
  • Parameters:
    'command' (string) - The command to execute (e.g., "bold", "italic").
    'showUI' (boolean, optional) - Whether to show the default user interface (deprecated).
    'value' (string, optional) - The value to use for the command.
  • Return Value: true if the command was executed successfully, false otherwise.
                    
                        document.execCommand('bold');

Q 26. document.adoptNode()

  • Description: Adopts a node from another document.
  • Syntax: document.adoptNode(node)
  • Parameters: node - The node to adopt.
  • Return Value: The adopted node.
                    
                        var otherDocument = document.implementation.createHTMLDocument('Other Document');
                        var node = otherDocument.createElement('div');
                        document.adoptNode(node);
                        document.body.appendChild(node);

Q 27. document.importNode()

  • Description: Imports a node from another document.
  • Syntax: document.importNode(node, deep)
  • Parameters:
    'node' - The node to import.
    'deep' (boolean) - Whether to import the node and all its descendants.
  • Return Value: The imported node
                    
                        var otherDocument = document.implementation.createHTMLDocument('Other Document');
                        var node = otherDocument.createElement('div');
                        var importedNode = document.importNode(node, true);
                        document.body.appendChild(importedNode);

Q 28. document.caretPositionFromPoint()

  • Description: Returns a CaretPosition object with the text offset and node at the specified coordinates.
  • Syntax: document.caretPositionFromPoint(x, y)
  • Parameters:
    'x' (number) - The horizontal coordinate.
    'y' (number) - The vertical coordinate.
  • Return Value: CaretPosition object or null
                    
                        var caretPosition = document.caretPositionFromPoint(100, 100);
                        if (caretPosition) {
                            console.log(caretPosition.offsetNode);
                            console.log(caretPosition.offset);
                        }
